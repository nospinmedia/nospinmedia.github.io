const today = est.toISOString().split("T")[0]; for (let s of slots) { const dt = new Date(`${today}T${s}:00-05:00`); if (dt > est) return dt; } // All today's slots passed â†’ first slot tomorrow const tomorrow = new Date(est.getTime() + 86400000); i < stories.length; i++) { stories[i].scheduled = currentSlot.toISOString(); const estDate = new Date(currentSlot.toLocaleString("en-US", { timeZone: "America/New_York" })); let slotIndex = slots.indexOf(estDate.toISOString().substring(11, 16)); const dateGroup = document.createElement("div"); dateGroup.className = "date-group"; dateGroup.innerHTML = `
